import turtle as t
from math import sin
from math import pi
t.shape('turtle')
t.speed(2)

n=8
p=2*pi/360
for i in range(n):
    t.penup()
    t.goto(20*(i+1), 0)
    t.pendown()
    a=40*(i+1)*sin(p*360/(2*(i+3)))
    t.left(90+180/(i+3))
    for j in range (i+3):
        t.forward(a)
        t.left(360/(i+3))
    t.right(90+180/(i+3))

input()
